2. Create build job (name: release)

pipeline {
    agent any

    stages {
        stage("release") {
             steps {
                  bat '''
                  cd ../CI-CD/builders/
                  mvn package
                  '''
             }
        }
    }
}

3. Parameterize job to use Maven or Gradle

pipeline {
    agent any
    parameters {
        choice(
        choices: ['Maven' , 'Gradle'],
        description: 'Project Management tool',
        name: 'TOOL')
    }
    stages {
        stage('release with Maven') {
            when {
                expression { return params.TOOL == 'Maven' }
            }
            steps {
                bat '''
                  cd ../CI-CD/builders/
                  mvn package
                '''
            }
        }
        stage('release with Gradle') {
            when {
                expression { return params.TOOL == 'Gradle' }
            }
            steps {
                bat '''
                  cd ../CI-CD/builders/utils
                  gradle build
                  cd ../services
                  gradle build
                  cd ../admin
                  gradle build
                  cd ../web
                  gradle build
                '''
            }
        }
    }
}
